Moving Average ---------- function movingAverage(arr, period) { let result = []; for (let i = 0; i <= arr.length - period; i++) { let sum = 0; for (let j = 0; j < period; j++) sum += arr[i + j]; result.push(sum / period); } return result; } // ---------- 8.
